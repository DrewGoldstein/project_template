plot_y <- function (data, variable) {
  plot <- ggplot(data = data,
    mapping = aes(x=ViolentCrimesPerPop, y = variable)) +
  geom_point() + 
  xlim(0.0,1.0) + 
  ylim(0.0,1.0) +
  labs(title = "Population against ViolentCrimesPerPop", 
       x = "ViolentCrimesPerPop",
       y= variable)
  print(plot)

}
